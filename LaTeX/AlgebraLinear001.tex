\begin{enumerate}
\item \textbf{Operações com Escalares}

$\alpha(\beta A) = (\alpha \beta) A$

$\alpha (A B) = (\alpha A) B = A (\alpha B)$

$\alpha (A \pm B) = \alpha A \pm \alpha B$

$(\alpha \pm \beta) A = \alpha A \pm \beta A$ \\


\item \textbf{Produto de Matrizes}

$A + B = B + A$

$(A + B) + C = A + (B + C)$

$A(BC) = A(BC)$

$(B + C) A = BA + CA$ \\

\end{enumerate}
